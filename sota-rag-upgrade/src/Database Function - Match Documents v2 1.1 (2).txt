-- (Optional) Keep the original vector-only search function if needed, updated for documents_v2
-- Renamed function to reflect table version
create or replace function match_documents_v2_vector (
  query_embedding vector(1536),
  match_count int default 10, -- Added default value
  filter jsonb DEFAULT '{}'
) returns table (
  id bigint,
  content text,
  metadata jsonb,
  similarity float
)
language plpgsql
as $$
#variable_conflict use_column
begin
  return query
  select
    dv2.id, -- Using alias dv2 for documents_v2
    dv2.content,
    dv2.metadata,
    -- Similarity calculation (1 = most similar)
    1 - (dv2.embedding <=> query_embedding) as similarity
  from documents_v2 as dv2 -- Updated table name
  where dv2.metadata @> filter
  order by dv2.embedding <=> query_embedding asc -- Order by distance (ascending)
  limit match_count;
end;
$$;
